## 3. 数据同步

### 3.1 基础同步
```javascript
"sync_skill": {
    content: async function(){
        // 同步玩家数据(联机模式下全局调用的数据均需同步)
        game.broadcastAll(function(player, num){
            player.storage.count = num;
        }, player, 1);
        
        // 等待同步完成
        await game.asyncBroadcast();
        
        // 同步动画效果
        game.broadcastAll(function(player){
            player.$draw();
        }, player);
        
        // 延迟等待动画
        game.delayx();
    }
}
```

### 3.2 复杂同步
```javascript
"complex_sync": {
    content: async function(){
        // 同步多个操作
        game.broadcastAll(function(player, cards){
            player.gain(cards);
            player.storage.gained = true;
        }, player, event.cards);
        
        // 等待所有玩家完成
        await game.asyncBroadcast();
        
        // 继续执行
        player.draw();
    }
}
```

## 4. 事件处理

### 4.1 事件同步
```javascript
"event_skill": {
    trigger: {player: 'phaseBegin'},
    content: async function(){
        // 创建同步事件
        var next = game.createEvent('customEvent');
        next.player = player;
        next.setContent(function(){
            // 事件内容
            player.draw();
        });
        
        // 等待事件完成
        await next;
    }
}
```

### 4.2 选择同步
```javascript
"choice_skill": {
    content: async function(){
        // 同步选择结果
        let result = await player.chooseControl('选项1', '选项2')
            .set('prompt', '请选择一个选项')
            .set('ai', ()=>{
                return '选项1';
            })
            .forResult();
            
        // 广播选择结果
        game.broadcastAll(function(player, choice){
            player.storage.choice = choice;
        }, player, result.control);
    }
}
```

"online_skill": {
    enable: "phaseUse",
    usable: 1,
    filter: function(event, player){
        return player.countCards('h') > 0;

    content: async function(){
        // 选择目标
        let result = await player.chooseTarget('选择一名角色')
            .set('ai', target => get.attitude(player, target))
            .forResult();

            let target = result.targets[0];
            
            // 同步选择结果
            game.broadcastAll(function(player, target){
                player.line(target);
            }, player, target);
            
            // 处理效果
            await target.draw();
            
            // 同步状态
            game.broadcast(function(player, target){
                player.storage.used = true;
                target.update();
            }, player, target);
        }
    }
}

1. 创建一个复杂联机技能：
   - 实现多步骤同步
   - 处理资源加载
   - 优化性能

<details>
<summary>参考答案 | 🟩 Easy</summary>

```javascript
"complex_online": {
    // 预加载资源
    preload: function(){
        game.preloadImage([
            'extension/我的扩展/image/effect1.jpg',
            'extension/我的扩展/image/effect2.jpg'
        ]);
    },
    
    enable: "phaseUse",
    content: async function(){
        // 选择牌
        let cards = await player.chooseCard('h', true).forResult();
        
        // 同步选择结果
        game.broadcastAll(function(player, cards){
            player.showCards(cards);
        }, player, cards.cards);
        
        await game.asyncBroadcast();

        // 选择效果
        let choice = await player.chooseControl('效果1', '效果2')
            .set('prompt', '请选择一个效果')
            .forResult();
            
        // 同步选择
        game.broadcastAll(function(player, choice){
            player.popup(choice.control);
        }, player, choice);
        
        // 执行效果
        if(result.control === '效果1'){
            await player.draw(2);
        } else {
            await player.recover();
        }
        
        // 批量同步状态
        game.broadcastAll(function(player){
            player.update();
            player.$draw();
        }, player);
    }
}
```
</details>
